Token Sync

This widget allows you to sync this website with your favorite design program.

    Integration Code

    import { SyncClient, pluginLinks, createCollection } from 'token-sync';
import type { TokenSyncPayload } from 'token-sync';

const SYNC_SERVER_URL = 'ws://localhost:8080';

let syncClient: SyncClient<TokenSyncPayload> | null = null;
let sessionToken: string | null = null;

// Initialize the sync client
function initSync() {
  syncClient = new SyncClient<TokenSyncPayload>({
    serverUrl: SYNC_SERVER_URL,
    clientType: 'web',
    origin: 'My Design System',
    onPaired: (token) => {
      sessionToken = token;
      updateWidgetStatus('ready', token);
    },
    onTargetConnected: () => {
      updateWidgetStatus('syncing', sessionToken ?? undefined);
    },
    onDisconnected: () => {
      updateWidgetStatus('disconnected');
    },
    onError: (error) => {
      updateWidgetStatus('error', undefined, error);
    },
  });

  syncClient.connect();
}

// Sync tokens to design tools
function syncTokens(color: string) {
  if (!syncClient) return;
  
  const payload = createCollection('colors', {
    primary: color,
  });
  
  syncClient.send(payload);
}

initSync();
